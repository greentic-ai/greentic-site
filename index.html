curl -X POST https://api.greentic.ai/workers \
  -H "authorization: Bearer $TOKEN" \
  -H "content-type: application/json" \
  -d '{"name":"invoice-bot","skills":["email.read","xero.create-invoice"],"flow":"ap-match"}'
import { Greentic } from "@greentic/sdk";
const g = new Greentic(process.env.GREENTIC_KEY);
await g.workers.create({ name: "noc-triage", skills: ["teams.read","servicenow.create"], flow: "noc" });
use greentic::Client;
let c = Client::new(std::env::var("GREENTIC_KEY")?);
c.workers().create("hr-onboarding")
 .skills(["email.read","workday.provision"])
 .flow("onboarding")
 .send().await?;
